Data Example — SQL

-- Top products by revenue (quarter)

  product_id,
  product_name,
  SUM(revenue) AS revenue_qtr
FROM sales
WHERE sale_date >= '2025-10-01' AND sale_date < '2025-12-01'
GROUP BY product_id, product_name
ORDER BY revenue_qtr DESC
LIMIT 10;
