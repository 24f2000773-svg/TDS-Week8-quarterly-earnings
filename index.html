Data example — SQL (Revenue by product)

-- Top products by revenue (Q4)
SELECT
  product_id,
  product_name,
  SUM(revenue) AS revenue_q4
FROM sales
WHERE sale_date BETWEEN '2025-07-01' AND '2025-09-30'
GROUP BY product_id, product_name
ORDER BY revenue_q4 DESC
LIMIT 10;
